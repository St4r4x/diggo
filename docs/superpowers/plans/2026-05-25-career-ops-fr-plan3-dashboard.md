# career-ops-fr — Plan 3: Application Tracker Terminal Dashboard

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a terminal dashboard (TUI) in Go to track job applications across the full hiring pipeline. The dashboard provides three views: a kanban pipeline, a full application detail form, and a statistics summary. All data is persisted to SQLite. The app is a standalone Go module living in `dashboard/` inside the existing `career-ops-fr` project.

**Architecture:** `dashboard/` is a self-contained Go module (`career-ops-fr/dashboard`). SQLite persists data at `../data/applications.db` (relative to the `dashboard/` directory, i.e. `career-ops-fr/data/applications.db`). Bubble Tea drives the TUI event loop. Lip Gloss handles all terminal styling. Tests cover `model/` and `db/` packages with in-memory SQLite (`:memory:`); UI packages get smoke tests that call `View()` directly without running a full Bubble Tea program. All Go commands use the full binary path `/usr/local/go/bin/go` since PATH may not include it.

**Tech Stack:** Go 1.21+, Bubble Tea v0.26.6, Lip Gloss v0.10.0, Bubbles v0.18.0, go-sqlite3 v1.14.22 (CGO), SQLite3 (via mattn/go-sqlite3)

---

## Application states (9 total, ordered)

```
À envoyer → Envoyée → Relance → Entretien RH → Entretien tech → Offre → Acceptée / Refusée / Abandonnée
```

Alert rule: any application in status `Envoyée` for more than 7 days since `send_date` is flagged with orange/yellow in the pipeline view.

// dashboard/model/application.go

// Package model defines the core data types for the career-ops-fr application tracker.
package model

import "time"

// FollowUpAlertDays is the number of days an application may remain in "Envoyée"
// status before it is flagged as stale in the pipeline view.
const FollowUpAlertDays = 7

// ValidStatuses is the ordered list of all nine pipeline states.
// The slice order mirrors the hiring pipeline left-to-right.
// Terminal states (no further action expected) are the last three elements.
var ValidStatuses = []string{
	"À envoyer",
	"Envoyée",
	"Relance",
	"Entretien RH",
	"Entretien tech",
	"Offre",
	"Acceptée",
	"Refusée",
	"Abandonnée",
}

// Application represents one tracked job application.
//
// SendDate and FollowUpDate use time.Time; a zero value (time.Time{}) means the
// field has not been set. CVPath, CoverLetterPath, and Notes are plain strings
// that default to empty. Contacts is a raw JSON string (array of objects with
// keys: name, linkedin, email) stored and retrieved verbatim from SQLite.
type Application struct {
	// ID is the SQLite AUTOINCREMENT primary key. Zero means not yet persisted.
	ID int64

	// Required fields.
	Company       string
	Role          string
	OfferURL      string
	DetectionDate time.Time
	ScoreGrade    string  // "A" | "B" | "C" | "D" | "F"
	ScoreValue    float64 // 0.0 – 5.0
	Status        string  // must be one of ValidStatuses

	// Optional fields — zero value means not set.
	SendDate        time.Time // date the application was sent
	Contacts        string    // JSON: [{"name":"...","linkedin":"...","email":"..."}]
	Notes           string
	CVPath          string
	CoverLetterPath string
	FollowUpDate    time.Time // next planned follow-up date
}

// IsStale reports whether the application has been sitting in "Envoyée" status
// for more than FollowUpAlertDays days as of the given reference time.
//
// Returns false if:
//   - Status is not "Envoyée"
//   - SendDate is the zero value (application not yet sent)
//   - The elapsed days since SendDate is <= FollowUpAlertDays
func (a Application) IsStale(now time.Time) bool {
	if a.Status != "Envoyée" {
		return false
	}
	if a.SendDate.IsZero() {
		return false
	}
	elapsed := now.Sub(a.SendDate)
	return elapsed > time.Duration(FollowUpAlertDays)*24*time.Hour
}

// openMemDB opens an in-memory SQLite database, creates the schema, and returns
// the *db.DB wrapper. The caller must close it after the test.
func openMemDB(t *testing.T) *db.DB {
	t.Helper()
	database, err := db.Open(":memory:")
	if err != nil {
		t.Fatalf("db.Open(:memory:) error: %v", err)
	}
	if err := database.CreateTable(); err != nil {
		t.Fatalf("CreateTable error: %v", err)
	}
	return database
}

func TestCreateTable(t *testing.T) {
	t.Run("creates applications table", func(t *testing.T) {
		rawDB, err := sql.Open("sqlite3", ":memory:")
		if err != nil {
			t.Fatalf("sql.Open error: %v", err)
		}
		defer rawDB.Close()

		database := db.Wrap(rawDB)
		if err := database.CreateTable(); err != nil {
			t.Fatalf("CreateTable error: %v", err)
		}

		var name string
		err = rawDB.QueryRow(
			"SELECT name FROM sqlite_master WHERE type='table' AND name='applications'",
		).Scan(&name)
		if err != nil {
			t.Fatalf("table not found after CreateTable: %v", err)
		}
		if name != "applications" {
			t.Errorf("table name = %q, want %q", name, "applications")
		}
	})

	t.Run("idempotent — second call does not error", func(t *testing.T) {
		database := openMemDB(t)
		defer database.Close()
		if err := database.CreateTable(); err != nil {
			t.Errorf("second CreateTable call returned error: %v", err)
		}
	})

// Open opens (or creates) the SQLite database file at path and returns a *DB.

func Open(path string) (*DB, error) {
	sqlDB, err := sql.Open("sqlite3", path)
	if err != nil {
		return nil, fmt.Errorf("db.Open: %w", err)
	}
	return &DB{sqlDB: sqlDB}, nil
}

// Wrap constructs a *DB from an existing *sql.DB. Used in tests where the caller
// manages the underlying connection.

	return &DB{sqlDB: sqlDB}

// CreateTable creates the applications table if it does not already exist.

		return fmt.Errorf("CreateTable: %w", err)

	return nil

// timeToNull converts a time.Time to a *string for SQLite storage.
// A zero time becomes nil (NULL). Non-zero times are stored as ISO 8601 date strings.
func timeToNull(t time.Time) *string {
	if t.IsZero() {
		return nil
	}
	s := t.UTC().Format("2006-01-02")
	return &s
}

// nullToTime converts a *string read from SQLite back to time.Time.
// nil (NULL) becomes the zero time.Time. Non-nil strings are parsed as ISO 8601.
func nullToTime(s *string) (time.Time, error) {
	if s == nil {
		return time.Time{}, nil
	}
	t, err := time.Parse("2006-01-02", *s)
	if err != nil {
		return time.Time{}, fmt.Errorf("nullToTime: parse %q: %w", *s, err)
	}
	return t, nil
}

// Insert inserts a new application row and returns the assigned primary key id.
// The app.ID field is ignored; SQLite assigns the id via AUTOINCREMENT.
func (d *DB) Insert(app model.Application) (int64, error) {
	res, err := d.sqlDB.Exec(
		`INSERT INTO applications
		    (company, role, offer_url, detection_date, score_grade, score_value,
		     status, send_date, contacts, notes, cv_path, cover_letter_path, follow_up_date)
		 VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
		app.Company,
		app.Role,
		app.OfferURL,
		app.DetectionDate.UTC().Format("2006-01-02"),
		app.ScoreGrade,

		app.Notes,
		app.CVPath,
		app.CoverLetterPath,
		timeToNull(app.FollowUpDate),
	)

	}
	id, err := res.LastInsertId()
	if err != nil {
		return 0, fmt.Errorf("Insert LastInsertId: %w", err)

	return id, nil
}

func rowToApplication(
	scanFn func(dest ...any) error,
) (model.Application, error) {
	var (
		app             model.Application
		sendDateStr     *string
		followUpDateStr *string
	)
	err := scanFn(
		&app.ID,
		&app.Company,

		&app.OfferURL,

		&app.ScoreGrade,

		&sendDateStr,
		&app.Contacts,
		&app.Notes,
		&app.CVPath,
		&app.CoverLetterPath,
		&followUpDateStr,
	)
	if err != nil {
		return model.Application{}, err

		return model.Application{}, fmt.Errorf("rowToApplication follow_up_date: %w", err)
	} else {
		app.FollowUpDate = fud
	}
	return app, nil
}

import (

// --- Lip Gloss styles ---

var (
	pipelineColumnStyle = lipgloss.NewStyle().
				Border(lipgloss.RoundedBorder()).
				BorderForeground(lipgloss.Color("63")).
				Padding(0, 1).
				Width(22)

	pipelineColumnHeaderStyle = lipgloss.NewStyle().
					Bold(true).
					Foreground(lipgloss.Color("63")).
					MarginBottom(1)

	cardStyle = lipgloss.NewStyle().
			Border(lipgloss.NormalBorder()).
			BorderForeground(lipgloss.Color("240")).
			Padding(0, 1).
			MarginBottom(1).
			Width(18)

	cardSelectedStyle = lipgloss.NewStyle().
				Border(lipgloss.NormalBorder()).
				BorderForeground(lipgloss.Color("212")).
				Padding(0, 1).
				MarginBottom(1).
				Width(18)

	cardStaleStyle = lipgloss.NewStyle().
			Border(lipgloss.NormalBorder()).
			BorderForeground(lipgloss.Color("214")). // orange
			Foreground(lipgloss.Color("214")).
			Padding(0, 1).
			MarginBottom(1).
			Width(18)

	gradeGoodStyle = lipgloss.NewStyle().Foreground(lipgloss.Color("82"))  // green
	gradeBadStyle  = lipgloss.NewStyle().Foreground(lipgloss.Color("226")) // yellow

	staleMarkerStyle = lipgloss.NewStyle().
				Foreground(lipgloss.Color("214")).
				Bold(true)

// pipelineKeyMap defines keybindings for the pipeline view.
type pipelineKeyMap struct {
	Left   key.Binding
	Right  key.Binding
	Up     key.Binding
	Down   key.Binding
	Select key.Binding
}

var pipelineKeys = pipelineKeyMap{
	Left:   key.NewBinding(key.WithKeys("left", "h")),
	Right:  key.NewBinding(key.WithKeys("right", "l")),
	Up:     key.NewBinding(key.WithKeys("up", "k")),
	Down:   key.NewBinding(key.WithKeys("down", "j")),
	Select: key.NewBinding(key.WithKeys("enter")),
}

// PipelineModel is the Bubble Tea model for the kanban pipeline view.
// It renders 9 columns (one per ValidStatus) with application cards.
type PipelineModel struct {
	apps         []model.Application
	colIdx       int   // which column is focused
	cardIdx      int   // which card within the focused column is selected
	selectedID   int64 // ID of the currently selected application (0 = none)
	width        int
	height       int
	now          time.Time
}

// NewPipelineModel constructs a PipelineModel with the given application list
// and terminal dimensions.
func NewPipelineModel(apps []model.Application, width, height int) PipelineModel {
	return PipelineModel{
		apps:   apps,
		width:  width,
		height: height,
		now:    time.Now(),
	}
}

// SelectedAppID returns the ID of the application under the cursor, or 0 if
// no application is selected.
func (m PipelineModel) SelectedAppID() int64 {
	return m.selectedID
}

// appsForStatus returns all applications in the given status, preserving order.
func (m PipelineModel) appsForStatus(status string) []model.Application {
	var result []model.Application
	for _, app := range m.apps {
		if app.Status == status {
			result = append(result, app)
		}
	}
	return result
}

// Init satisfies the tea.Model interface. No I/O needed on startup.
func (m PipelineModel) Init() tea.Cmd {
	return nil
}

// Update handles keyboard events for column and card navigation.
func (m PipelineModel) Update(msg tea.Msg) (PipelineModel, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch {
		case key.Matches(msg, pipelineKeys.Left):
			if m.colIdx > 0 {
				m.colIdx--
				m.cardIdx = 0
				m.selectedID = 0
			}
		case key.Matches(msg, pipelineKeys.Right):
			if m.colIdx < len(model.ValidStatuses)-1 {
				m.colIdx++
				m.cardIdx = 0
				m.selectedID = 0
			}
		case key.Matches(msg, pipelineKeys.Up):
			if m.cardIdx > 0 {
				m.cardIdx--
			}
		case key.Matches(msg, pipelineKeys.Down):
			colApps := m.appsForStatus(model.ValidStatuses[m.colIdx])
			if m.cardIdx < len(colApps)-1 {
				m.cardIdx++
			}
		case key.Matches(msg, pipelineKeys.Select):
			colApps := m.appsForStatus(model.ValidStatuses[m.colIdx])
			if m.cardIdx < len(colApps) {
				m.selectedID = colApps[m.cardIdx].ID
			}
		}
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
	}
	return m, nil
}

// renderCard renders a single application card as a styled string.
func (m PipelineModel) renderCard(app model.Application, selected bool) string {
	stale := app.IsStale(m.now)

	// Grade color
	var gradeStr string
	if app.ScoreGrade == "A" || app.ScoreGrade == "B" {
		gradeStr = gradeGoodStyle.Render(fmt.Sprintf("%s %.1f", app.ScoreGrade, app.ScoreValue))
	} else {
		gradeStr = gradeBadStyle.Render(fmt.Sprintf("%s %.1f", app.ScoreGrade, app.ScoreValue))
	}

	// Stale marker
	staleStr := ""
	if stale {
		staleStr = staleMarkerStyle.Render(" ! RELANCE")
	}

	content := fmt.Sprintf("%s%s\n%s\n%s", app.Company, staleStr, app.Role, gradeStr)

	switch {
	case selected:
		return cardSelectedStyle.Render(content)
	case stale:
		return cardStaleStyle.Render(content)
	default:
		return cardStyle.Render(content)
	}
}

// View renders the full kanban board as a string.
// Each status gets one column. The focused column is highlighted.
// If the terminal width is too small to show all columns, only the columns
// that fit are shown (starting from the focused column).
func (m PipelineModel) View() string {
	// Compute how many columns fit given the current terminal width.
	// Each column is ~24 chars wide (22 content + 2 border).
	colWidth := 24
	visibleCols := m.width / colWidth
	if visibleCols < 1 {
		visibleCols = 1
	}
	if visibleCols > len(model.ValidStatuses) {
		visibleCols = len(model.ValidStatuses)
	}

	// Centre the visible window around the focused column.
	startCol := m.colIdx - visibleCols/2
	if startCol < 0 {
		startCol = 0
	}
	if startCol+visibleCols > len(model.ValidStatuses) {
		startCol = len(model.ValidStatuses) - visibleCols
	}

	var columns []string
	for i := startCol; i < startCol+visibleCols; i++ {
		status := model.ValidStatuses[i]
		colApps := m.appsForStatus(status)

		// Header
		header := pipelineColumnHeaderStyle.Render(
			fmt.Sprintf("%s (%d)", status, len(colApps)),
		)

		// Cards
		var cardLines []string
		cardLines = append(cardLines, header)
		for j, app := range colApps {
			selected := (i == m.colIdx && j == m.cardIdx)
			cardLines = append(cardLines, m.renderCard(app, selected))
		}

		colContent := ""
		for _, line := range cardLines {
			colContent += line + "\n"
		}

		// Highlight focused column border
		style := pipelineColumnStyle
		if i == m.colIdx {
			style = pipelineColumnStyle.Copy().BorderForeground(lipgloss.Color("212"))
		}
		columns = append(columns, style.Render(colContent))
	}

	return lipgloss.JoinHorizontal(lipgloss.Top, columns...)
}

// SaveMsg is sent on the Bubble Tea message bus when the user saves a form.
// The parent model (AppModel) handles persistence.
type SaveMsg struct {

}

// CancelMsg is sent when the user presses Esc in the detail view.
type CancelMsg struct{}

// --- field indices ---

	fieldRole
	fieldOfferURL
	fieldDetectionDate

	fieldScoreValue

	Foreground(lipgloss.Color("243")).

	Foreground(lipgloss.Color("255"))

func NewDetailModel(app model.Application, isNew bool) DetailModel {

		isNew: isNew,

	}

	labels := [fieldCount]string{
		"Company",
		"Role",
		"Offer URL",

		}

	}

// formatDate converts a time.Time to "YYYY-MM-DD". Zero time returns "".
func formatDate(t time.Time) string {
	if t.IsZero() {
		return ""
	}
	return t.UTC().Format("2006-01-02")
}

// formatFloat formats a float64 as a string, omitting trailing zeros.
// Zero value is returned as "".
func formatFloat(f float64) string {
	if f == 0 {
		return ""
	}
	return strconv.FormatFloat(f, 'f', -1, 64)
}

// IsNew reports whether this form is for a new (unsaved) application.
func (m DetailModel) IsNew() bool {
	return m.isNew
}

// GetApplication returns the model.Application built from the current form values.
// Parsing errors are silently ignored — use Validate() before calling Save.
func (m DetailModel) GetApplication() model.Application {
	app := m.app
	app.Company = m.inputs[fieldCompany].Value()
	app.Role = m.inputs[fieldRole].Value()
	app.OfferURL = m.inputs[fieldOfferURL].Value()
	app.DetectionDate, _ = parseDate(m.inputs[fieldDetectionDate].Value())
	app.ScoreGrade = m.inputs[fieldScoreGrade].Value()
	app.ScoreValue, _ = strconv.ParseFloat(m.inputs[fieldScoreValue].Value(), 64)
	app.Status = m.inputs[fieldStatus].Value()
	app.SendDate, _ = parseDate(m.inputs[fieldSendDate].Value())
	app.FollowUpDate, _ = parseDate(m.inputs[fieldFollowUpDate].Value())
	app.CVPath = m.inputs[fieldCVPath].Value()
	app.CoverLetterPath = m.inputs[fieldCoverLetterPath].Value()
	app.Notes = m.inputs[fieldNotes].Value()
	return app
}

// parseDate parses an ISO date string "YYYY-MM-DD". Returns zero time if s is empty.
func parseDate(s string) (time.Time, error) {
	s = strings.TrimSpace(s)
	if s == "" {
		return time.Time{}, nil
	}
	return time.Parse("2006-01-02", s)
}

// Validate checks that all required fields are filled and have valid formats.
// Returns an error message string, or "" if valid.
func (m DetailModel) Validate() string {
	if strings.TrimSpace(m.inputs[fieldCompany].Value()) == "" {
		return "Company is required"
	}
	if strings.TrimSpace(m.inputs[fieldRole].Value()) == "" {
		return "Role is required"
	}
	if strings.TrimSpace(m.inputs[fieldDetectionDate].Value()) == "" {
		return "Detection date is required"
	}
	if _, err := parseDate(m.inputs[fieldDetectionDate].Value()); err != nil {
		return fmt.Sprintf("Detection date: %v", err)
	}
	if strings.TrimSpace(m.inputs[fieldScoreGrade].Value()) == "" {
		return "Score grade is required"
	}
	svStr := strings.TrimSpace(m.inputs[fieldScoreValue].Value())
	if svStr != "" {
		if _, err := strconv.ParseFloat(svStr, 64); err != nil {
			return "Score value must be a number (e.g. 4.5)"
		}
	}
	statusVal := strings.TrimSpace(m.inputs[fieldStatus].Value())
	validStatus := false
	for _, s := range model.ValidStatuses {
		if s == statusVal {
			validStatus = true
			break
		}
	}
	if !validStatus {
		return fmt.Sprintf("Invalid status %q. Valid: %s", statusVal, strings.Join(model.ValidStatuses, ", "))
	}
	if sdStr := strings.TrimSpace(m.inputs[fieldSendDate].Value()); sdStr != "" {
		if _, err := parseDate(sdStr); err != nil {
			return fmt.Sprintf("Send date: %v", err)
		}
	}
	if fudStr := strings.TrimSpace(m.inputs[fieldFollowUpDate].Value()); fudStr != "" {
		if _, err := parseDate(fudStr); err != nil {
			return fmt.Sprintf("Follow-up date: %v", err)
		}
	}
	return ""
}

// Init satisfies tea.Model. No I/O on startup.
func (m DetailModel) Init() tea.Cmd {
	return textinput.Blink
}

// Update handles keypresses for the detail form.
//
//   - Tab / Shift+Tab: cycle focus between fields.
//   - Enter on last field or Ctrl+S: validate and emit SaveMsg.
//   - Esc: emit CancelMsg.
//   - All other keys: delegate to the focused textinput.
func (m DetailModel) Update(msg tea.Msg) (DetailModel, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "esc":
			return m, func() tea.Msg { return CancelMsg{} }

		case "ctrl+s":
			if errMsg := m.Validate(); errMsg != "" {
				m.errorMsg = errMsg
				return m, nil
			}
			return m, func() tea.Msg { return SaveMsg{App: m.GetApplication()} }

		case "tab":
			m.inputs[m.focusIdx].Blur()
			m.focusIdx = (m.focusIdx + 1) % fieldCount
			m.inputs[m.focusIdx].Focus()
			return m, textinput.Blink

		case "shift+tab":
			m.inputs[m.focusIdx].Blur()
			m.focusIdx = (m.focusIdx - 1 + fieldCount) % fieldCount
			m.inputs[m.focusIdx].Focus()
			return m, textinput.Blink
		}
	}

	// Delegate to the focused input.
	var cmd tea.Cmd
	m.inputs[m.focusIdx], cmd = m.inputs[m.focusIdx].Update(msg)
	return m, cmd
}

// View renders the detail form as a string.
func (m DetailModel) View() string {
	var sb strings.Builder

	title := "New Application"
	if !m.isNew {
		title = fmt.Sprintf("Edit Application  #%d — %s", m.app.ID, m.app.Company)
	}
	sb.WriteString(detailTitleStyle.Render(title) + "\n")

		label := detailLabelStyle.Render(fieldLabels[i] + ":")

	if m.errorMsg != "" {

				Foreground(lipgloss.Color("63")).

				BorderStyle(lipgloss.NormalBorder()).
				BorderForeground(lipgloss.Color("240"))

	statsTableRowAltStyle = lipgloss.NewStyle().

	statsZeroCountStyle = lipgloss.NewStyle().
				Foreground(lipgloss.Color("240"))

// StatsModel is the Bubble Tea model for the statistics view.
type StatsModel struct {
	stats db.Stats
}

// NewStatsModel constructs a StatsModel from a db.Stats value.
func NewStatsModel(stats db.Stats) StatsModel {
	return StatsModel{stats: stats}
}

// Init satisfies tea.Model.

	return nil
}

// Update handles messages. The stats view has no user-interaction beyond scrolling
// (which is handled by the parent AppModel). Most messages are ignored here.

	return m, nil
}

// View renders the statistics summary as a string.
func (m StatsModel) View() string {
	var sb strings.Builder

	sb.WriteString(statsHeadingStyle.Render("Pipeline Statistics") + "\n\n")

	// Key metrics table (label + value pairs).
	metrics := []struct {
		label string
		value string
		stale bool
	}{
		{"Total applications", fmt.Sprintf("%d", m.stats.TotalApplications), false},
		{"Response rate", fmt.Sprintf("%.1f%%", m.stats.ResponseRate), false},
		{"Interviews obtained", fmt.Sprintf("%d", m.stats.InterviewCount), false},
		{"Stale follow-ups (>7d Envoyée)", fmt.Sprintf("%d", m.stats.StaleCount), m.stats.StaleCount > 0},
	}

	for _, metric := range metrics {
		label := statsLabelStyle.Render(metric.label + ":")
		var value string
		if metric.stale {
			value = statsStaleWarningStyle.Render(metric.value + "  !")
		} else {
			value = statsValueStyle.Render(metric.value)
		}
		sb.WriteString(fmt.Sprintf("  %s  %s\n", label, value))
	}

	sb.WriteString("\n")

	// Per-status breakdown table.
	sb.WriteString(statsTableHeaderStyle.Render(
		fmt.Sprintf("  %-22s  %s", "Status", "Count"),
	) + "\n")

	for i, status := range model.ValidStatuses {
		count := m.stats.ByStatus[status]
		countStr := fmt.Sprintf("%d", count)
		var rowStyle lipgloss.Style
		if count == 0 {
			countStr = statsZeroCountStyle.Render(countStr)
			rowStyle = statsZeroCountStyle
		} else if i%2 == 0 {
			rowStyle = statsTableRowStyle
		} else {
			rowStyle = statsTableRowAltStyle
		}
		row := fmt.Sprintf("  %-22s  %s", status, countStr)
		sb.WriteString(rowStyle.Render(row) + "\n")
	}

	return sb.String()
}

// dashboard/ui/app.go

// Package ui contains all Bubble Tea models for the career-ops-fr terminal dashboard.
//
// AppModel is the root model. It owns the current view state (pipeline / detail /
// stats) and routes keypresses to the appropriate sub-model. Database persistence
// is also triggered here in response to SaveMsg from DetailModel.
package ui

import (
	"fmt"
	"time"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/charmbracelet/lipgloss"

	"career-ops-fr/dashboard/db"
	"career-ops-fr/dashboard/model"
)

// view is an enum for the three top-level views.
type view int

const (
	viewPipeline view = iota
	viewDetail
	viewStats
)

// --- Tab bar styles ---

var (
	tabActiveStyle = lipgloss.NewStyle().
			Bold(true).
			Foreground(lipgloss.Color("212")).
			Background(lipgloss.Color("236")).
			Padding(0, 2)

	tabInactiveStyle = lipgloss.NewStyle().
				Foreground(lipgloss.Color("240")).
				Padding(0, 2)

	statusBarStyle = lipgloss.NewStyle().
			Foreground(lipgloss.Color("243")).
			Italic(true)

	errorBarStyle = lipgloss.NewStyle().
			Foreground(lipgloss.Color("196")).
			Bold(true)
)

// AppModel is the root Bubble Tea model for the career-ops-fr dashboard.
//
// It owns:
//   - The active database connection (*db.DB).
//   - The current view (pipeline / detail / stats).
//   - Sub-model instances for each view.
//   - The status bar message shown at the bottom.
type AppModel struct {
	database     *db.DB
	currentView  view
	pipeline     PipelineModel
	detail       DetailModel
	stats        StatsModel
	width        int
	height       int
	statusMsg    string
	isError      bool
}

// NewAppModel constructs the root AppModel, loads all applications from the
// database, and initialises each sub-model.
func NewAppModel(database *db.DB) (AppModel, error) {
	apps, err := database.GetAll()
	if err != nil {
		return AppModel{}, fmt.Errorf("NewAppModel: %w", err)
	}

	stats, err := database.GetStats(time.Now())
	if err != nil {
		return AppModel{}, fmt.Errorf("NewAppModel GetStats: %w", err)
	}

	return AppModel{
		database:    database,
		currentView: viewPipeline,
		pipeline:    NewPipelineModel(apps, 80, 24),
		detail:      NewDetailModel(model.Application{}, true),
		stats:       NewStatsModel(stats),
		statusMsg:   "Ready  |  p=pipeline  s=stats  n=new  e=edit  q=quit",
	}, nil
}

// Init starts the Bubble Tea program. We send an initial reload command so the
// pipeline is refreshed immediately.
func (m AppModel) Init() tea.Cmd {
	return nil
}

// reloadApps fetches all applications from the database and refreshes the
// pipeline and stats sub-models.
func (m *AppModel) reloadApps() error {
	apps, err := m.database.GetAll()
	if err != nil {
		return fmt.Errorf("reloadApps GetAll: %w", err)
	}
	stats, err := m.database.GetStats(time.Now())
	if err != nil {
		return fmt.Errorf("reloadApps GetStats: %w", err)
	}
	m.pipeline = NewPipelineModel(apps, m.width, m.height-3) // subtract header+tabs+status bar
	m.stats = NewStatsModel(stats)
	return nil
}

// Update is the root message handler.
//
// Key routing:
//   - q / ctrl+c: quit
//   - tab / shift+tab: cycle views
//   - p: switch to pipeline
//   - s: switch to stats
//   - n: open blank detail form (new application)
//   - e: open detail form for the selected pipeline application
//   - All other keys: routed to the current sub-model
//
// Message routing:
//   - SaveMsg: persist the application and reload
//   - CancelMsg: return to pipeline
func (m AppModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {

	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
		updated, cmd := m.pipeline.Update(msg)
		m.pipeline = updated
		return m, cmd

	case tea.KeyMsg:
		switch msg.String() {
		case "q", "ctrl+c":
			return m, tea.Quit

		case "tab":
			m.currentView = (m.currentView + 1) % 3
			m.isError = false
			return m, nil

		case "shift+tab":
			m.currentView = (m.currentView + 2) % 3 // +2 mod 3 = -1 mod 3
			m.isError = false
			return m, nil

		case "p":
			m.currentView = viewPipeline
			m.isError = false
			return m, nil

		case "s":
			if err := m.reloadApps(); err != nil {
				m.statusMsg = err.Error()
				m.isError = true
			} else {
				m.currentView = viewStats
				m.isError = false
			}
			return m, nil

		case "n":
			m.detail = NewDetailModel(model.Application{}, true)
			m.currentView = viewDetail
			m.isError = false
			return m, textinput.Blink

		case "e":
			selID := m.pipeline.SelectedAppID()
			if selID == 0 {
				m.statusMsg = "No application selected — navigate the pipeline first (arrow keys + Enter)"
				m.isError = true
				return m, nil
			}
			app, err := m.database.GetByID(selID)
			if err != nil {
				m.statusMsg = fmt.Sprintf("Failed to load application id=%d: %v", selID, err)
				m.isError = true
				return m, nil
			}
			m.detail = NewDetailModel(app, false)
			m.currentView = viewDetail
			m.isError = false
			return m, textinput.Blink
		}

	case SaveMsg:
		app := msg.App
		var err error
		if app.ID == 0 {
			_, err = m.database.Insert(app)
			if err == nil {
				m.statusMsg = fmt.Sprintf("Saved new application: %s — %s", app.Company, app.Role)
			}
		} else {
			err = m.database.Update(app)
			if err == nil {
				m.statusMsg = fmt.Sprintf("Updated application #%d: %s — %s", app.ID, app.Company, app.Role)
			}
		}
		if err != nil {
			m.statusMsg = fmt.Sprintf("Save error: %v", err)
			m.isError = true
			return m, nil
		}
		m.isError = false
		if err := m.reloadApps(); err != nil {
			m.statusMsg = err.Error()
			m.isError = true
		}
		m.currentView = viewPipeline
		return m, nil

	case CancelMsg:
		m.currentView = viewPipeline
		m.isError = false
		m.statusMsg = "Cancelled"
		return m, nil
	}

	// Route to the active sub-model.
	var cmd tea.Cmd
	switch m.currentView {
	case viewPipeline:
		updated, c := m.pipeline.Update(msg)
		m.pipeline = updated
		cmd = c
	case viewDetail:
		updated, c := m.detail.Update(msg)
		m.detail = updated
		cmd = c
	case viewStats:
		updated, c := m.stats.Update(msg)
		m.stats = updated
		cmd = c
	}
	return m, cmd
}

// renderTabBar renders the three tab labels with the active one highlighted.
func (m AppModel) renderTabBar() string {
	tabs := []string{"Pipeline", "Detail", "Stats"}
	rendered := make([]string, len(tabs))
	for i, label := range tabs {
		if view(i) == m.currentView {
			rendered[i] = tabActiveStyle.Render(label)
		} else {
			rendered[i] = tabInactiveStyle.Render(label)
		}
	}
	return lipgloss.JoinHorizontal(lipgloss.Top, rendered...)
}

// View renders the full terminal UI: tab bar + active view content + status bar.
func (m AppModel) View() string {
	tabBar := m.renderTabBar()

	var content string
	switch m.currentView {
	case viewPipeline:
		content = m.pipeline.View()
	case viewDetail:
		content = m.detail.View()
	case viewStats:
		content = m.stats.View()
	}

	var statusLine string
	if m.isError {
		statusLine = errorBarStyle.Render(m.statusMsg)
	} else {
		statusLine = statusBarStyle.Render(m.statusMsg)
	}

	return lipgloss.JoinVertical(lipgloss.Left,
		tabBar,
		content,
		statusLine,
	)
}

// dashboard/main.go

// Command dashboard launches the career-ops-fr terminal application tracker.
//
// Usage:
//
//	/usr/local/go/bin/go run ./dashboard/
//
// The SQLite database is created at data/applications.db relative to the
// career-ops-fr project root (one directory above dashboard/).
package main

import (
	"fmt"
	"os"
	"path/filepath"
	"runtime"

	tea "github.com/charmbracelet/bubbletea"

	"career-ops-fr/dashboard/db"
	"career-ops-fr/dashboard/ui"
)

func main() {
	// Locate the data directory relative to the binary's source file location.
	// When run with `go run ./dashboard/`, the working directory is the project root.
	dbDir := filepath.Join("data")
	if err := os.MkdirAll(dbDir, 0o755); err != nil {
		fmt.Fprintf(os.Stderr, "cannot create data directory: %v\n", err)
		os.Exit(1)
	}

	dbPath := filepath.Join(dbDir, "applications.db")

	database, err := db.Open(dbPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "cannot open database %s: %v\n", dbPath, err)
		os.Exit(1)
	}
	defer database.Close()

	if err := database.CreateTable(); err != nil {
		fmt.Fprintf(os.Stderr, "cannot create schema: %v\n", err)
		os.Exit(1)
	}

	appModel, err := ui.NewAppModel(database)
	if err != nil {
		fmt.Fprintf(os.Stderr, "cannot init app model: %v\n", err)
		os.Exit(1)
	}

	// Use AltScreen for a full-terminal experience and enable mouse support.
	p := tea.NewProgram(appModel, tea.WithAltScreen(), tea.WithMouseCellMotion())

	_ = runtime.GOMAXPROCS(runtime.NumCPU())

	if _, err := p.Run(); err != nil {
		fmt.Fprintf(os.Stderr, "program error: %v\n", err)
		os.Exit(1)
	}
}
